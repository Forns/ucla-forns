Your Week 7 Classwork

In class, you (as a group) designed programs to match some example specifications. Those answers are listed herein.

Example One: stringArrayMatch

Design a function stringArrayMatch that takes as input one array of C++ strings, and one array of cstrings, and compares each of n elements between the two arrays. If the strings / cstrings at a given element are equal, change that element in both arrays to be the word "MATCH". 

  #include <iostream>
  #include <string>
  #include <cstring>
  using namespace std;
  
  const int MAX_CSTRING_LENGTH = 100;
  const int ARRAY_ROWS = 3;
  void stringArrayMatch (string s[], char c[][MAX_CSTRING_LENGTH], int n);
  
  int main () {
      string s[] = {"this", "func", "ROCKS"};
      char c[][MAX_CSTRING_LENGTH] = {"This", "func", "rocks"};
      stringArrayMatch(s, c, ARRAY_ROWS);
      
      for (int i = 0; i < 3; i++) {
          cout << "s[" << i << "]" << s[i] << endl;
          cout << "c[" << i << "]" << c[i] << endl;
      }
  }
  
  void stringArrayMatch (string s[], char c[][MAX_CSTRING_LENGTH], int n) {
      for (int i = 0; i < n; i++) {
          int len = strlen(c[i]);
          bool match = true;
          for (int j = 0; j < s[i].length(); j++) {
              if (j >= len) {
                  break;
              }
              if (s[i][j] != c[i][j]) {
                  match = false;
                  break;
              }
          }
          if (match) {
              s[i] = "MATCH";
              strcpy(c[i], "MATCH");
          }
      }
  }

Example Two: selfStrCat

Design a function selfStrcat that takes as input a single cstring and safely appends itself to itself. Assume that the argument has reserved enough room for this operation.

  #include <iostream>
  #include <cstring>
  #include <cctype>
  using namespace std;
  
  void selfStrcat (char c[]);
  
  int main () {
      char c[100] = "Finally, I'll work!";
      selfStrcat(c);
      
      // Prints: "Finally, I'll work!Finally, I'll work!"
      cout << c << endl;
      
      char t[100] = "";
      selfStrcat(t);
      
      // Prints: ""
      cout << t << endl;
      
      char devious[100] = "HA!\0HA!\0";
      selfStrcat(devious);
      
      // Prints: "HA!HA!"
      cout << devious << endl;
  }
  
  void selfStrcat (char c[]) {
      // Find the length until the first null
      int len = strlen(c);
      // Go through all len characters
      for (int i = 0; i < len; i++) {
          c[len + i] = c[i];
      }
      c[2 * len] = '\0';
  }
