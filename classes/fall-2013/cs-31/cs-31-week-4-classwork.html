Your Week 4 Classwork

In class, you (as a group) designed programs to match some example specifications. Those answers are listed herein.

Example One: Preset Substring

Design a function called presetSubstring that takes in a string s, and an int i, and then returns a string of all characters starting at the index of s designated by a global namespace constant int START up until index START + i. You may not use the builtin .substr(...) function!

Test Cases:
...
const int START = 1;
...
presetSubstring("test", 3); // Returns "est"
presetSubstring("testing", 3); // Returns "est"
presetSubstring("", 5); // Returns ""
presetSubstring("testing", -1); // Returns ""
presetSubstring("testing", 5000); // Returns "esting" 

  #include <iostream>
  #include <string>
  using namespace std;
  
  /*
   * Design a function called presetSubstring 
   * that takes in a string s, and an int length, 
   * and then returns a string of all characters 
   * starting at the index of s designated by a 
   * global namespace constant int START up until 
   * index START + i. You may not use the builtin 
   * .substr(...) function!
   */
  
  const int START = 1;
  string presetSubstring (string s, int length);
  
  int main () {
      // Tests
      cout << presetSubstring("test", 3) << endl; // Returns "est"
      cout << presetSubstring("testing", 3) << endl; // Returns "est"
      cout << presetSubstring("", 5) << endl; // Returns ""
      cout << presetSubstring("testing", -1) << endl; // Returns ""
      cout << presetSubstring("testing", 0) << endl; // Returns ""
      cout << presetSubstring("testing", 5000) << endl; // Returns "esting"
  }
  
  string presetSubstring (string s, int length) {
      string result;
      if (length < 0) {
          return "";
      }
      
      for (int i = START; i < s.size(); i++) {
          // Case where we've met our length criteria
          if (result.size() == length) {
              break;
          }
          result += s[i];
      }
      
      return result;
  }

Example Two: Even Letter Thief

Design a function called evenLetterThief of return type void that takes in a string by reference and then removes all of its even index letters.

  #include <iostream>
  #include <string>
  using namespace std;
  
  void evenLetterThief (string& s);
  
  int main () {
      string s = "test";
      evenLetterThief(s);
      cout << s << endl;
      
      s = "t";
      evenLetterThief(s);
      cout << s << endl;
      
      s = "";
      evenLetterThief(s);
      cout << s << endl;
  }
  
  void evenLetterThief (string& s) {
      for (int i = 0; i < s.length(); i += 2) {
          s = s.substr(0, i) + s.substr(i + 1, s.length());
          i--;
      }
  }
